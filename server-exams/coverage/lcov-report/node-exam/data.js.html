}

function resultToVisit (data) {
  const result = {};
  data.map((item) => {
    result[item.pathname] = true;
  });
  return result;
}
 
function resultData (data, element) {
    let $ = cheerio.load(data, config.cheerioLoad);
    const result = [];
    $('body').find(element).each((item, el) => {
      result.push($(el).html());
    });
   return result;
}
